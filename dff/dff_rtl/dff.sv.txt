module dff(
  input logic clk,
  input logic rst_n,
  input logic d,
  output logic  q
);
  
  always_ff@(posedge clk or negedge rst_n)
    begin
      if(!rst_n)
        q <=1'b0;
      else
        q <=d;
    end
  
  property reset;
    @(posedge clk)
    !rst_n |-> (q == 0);
  endproperty

  assert property(reset);

  property d_capture;
    @(posedge clk)
    disable iff(!rst_n)
    !$isunknown($past(d)) |-> (q == $past(d));
  endproperty

  assert property(d_capture);
  
endmodule
