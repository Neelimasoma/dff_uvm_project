class dff_scoreboard extends uvm_scoreboard;

  `uvm_component_utils(dff_scoreboard)

  uvm_analysis_imp#(dff_txn,dff_scoreboard)scb_imp;

  int total_txns;
  int pass_count,fail_count,rst_count;
  bit prev_d=0;

  function new(string name="dff_scoreboard",uvm_component parent=null);
    super.new(name,parent);
    scb_imp=new("scb_imp",this);
  endfunction

  function void write(dff_txn txn);

    total_txns++;

    if(txn.rst_n==0)
      begin
        rst_count++;
        prev_d=txn.d;
        return;
      end

    if(txn.q == prev_d)begin
      pass_count++;
      `uvm_info(get_type_name(),$sformatf("pass d=%0b and q=%0b(match)",txn.d,txn.q),UVM_LOW);
    end

    else
      begin
        fail_count++;
        `uvm_error(get_type_name(),$sformatf("fail d=%0b and q=%0b(fail)",txn.d,txn.q));
      end
    
    prev_d=txn.d;

  endfunction 

  function void report_phase(uvm_phase phase);
    `uvm_info(get_type_name(),$sformatf("total trans=%0d,pass count=%0d,fail_count=%0d,rst_count=%0d",total_txns,pass_count,fail_count,rst_count),UVM_LOW);
  endfunction

endclass
