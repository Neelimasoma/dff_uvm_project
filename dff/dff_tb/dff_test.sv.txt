  endfunction

    env=dff_env::type_id::create("env",this);
  endfunction

    phase.raise_objection(this);
    seq=dff_seq::type_id::create("seq");
    seq.start(env.agnt.seqr);

    phase.drop_objection(this);
  endtask

  endfunction
