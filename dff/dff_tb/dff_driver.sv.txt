class dff_driver extends uvm_driver#(dff_txn);
  `uvm_component_utils(dff_driver)

  virtual dff_intf vif;

  function new(string name="dff_driver",uvm_component parent=null);
    super.new(name,parent);
  endfunction

  function void build_phase(uvm_phase phase);
    super.build_phase(phase);
    if(!uvm_config_db#(virtual dff_intf)::get(this,"","vif",vif))
      `uvm_fatal(get_type_name(),"vif not set");
  endfunction

  task run_phase(uvm_phase phase);
    dff_txn txn;

    forever begin
      seq_item_port.get_next_item(txn);
      @(posedge vif.clk);
      vif.d      <=  txn.d;
      seq_item_port.item_done();

      `uvm_info(get_type_name(),$sformatf("driver d=%d",txn.d),UVM_LOW);
    end
  endtask

endclass
