class dff_seq extends uvm_sequence#(dff_txn);

  `uvm_object_utils(dff_seq);

  function new(string name="dff_seq");
    super.new(name);
  endfunction

  task body();
    dff_txn txn;

    `uvm_info(get_type_name(),"starting dff sequence",UVM_LOW)

    repeat(200)begin
      txn=dff_txn::type_id::create("txn");
      start_item(txn);
      assert(txn.randomize());
      finish_item(txn);
      `uvm_info(get_type_name(),$sformatf("generate txn %s",txn.convert2string()),UVM_LOW)
    end
  endtask 
endclass
